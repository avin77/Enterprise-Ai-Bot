```sql
-- Aurora PostgreSQL Serverless v2 schema for FAQ vector store
-- Run once: python scripts/run_ingest.py --setup

CREATE EXTENSION IF NOT EXISTS vector;

CREATE TABLE IF NOT EXISTS faq_chunks (
    id           SERIAL PRIMARY KEY,
    chunk_id     TEXT NOT NULL UNIQUE,          -- e.g. "jackson-faq-2024.pdf:chunk:0"
    text         TEXT NOT NULL,
    embedding    vector(1536),                  -- Bedrock Titan V2 output
    source_doc   TEXT NOT NULL,
    department   TEXT NOT NULL DEFAULT 'general',
    created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Full-text search index
CREATE INDEX IF NOT EXISTS faq_chunks_fts ON faq_chunks
    USING GIN (to_tsvector('english', text));

-- Vector similarity index (cosine distance via <=>)
CREATE INDEX IF NOT EXISTS faq_chunks_vector ON faq_chunks
    USING ivfflat (embedding vector_cosine_ops) WITH (lists = 50);
```

```python
from __future__ import annotations
import re
from pathlib import Path

try:
    import pdfplumber
except ImportError:
    pdfplumber = None  # type: ignore

_DEPT_MAP = {
    "permit": "planning", "zoning": "planning", "building": "planning",
    "tax": "finance", "payment": "finance", "vendor": "finance", "budget": "finance",
    "election": "elections", "voter": "elections",
    "emergency": "emergency", "flood": "emergency",
    "court": "courts", "fee": "courts",
    "road": "public-works", "traffic": "public-works",
    "recycl": "utilities", "waste": "utilities", "water": "utilities",
    "park": "parks", "recreation": "parks",
    "sheriff": "sheriff", "police": "sheriff",
    "hr": "hr", "employee": "hr", "benefit": "hr",
}

def _infer_department(filename: str) -> str:
    lower = filename.lower()
    for keyword, dept in _DEPT_MAP.items():
        if keyword in lower:
            return dept
    return "general"

def extract_chunks_from_pdf(
    pdf_path: str,
    chunk_words: int = 310,   # ~400 tokens at 1.3 tokens/word
    overlap_words: int = 38,  # ~50 tokens overlap
) -> list[dict]:
    if pdfplumber is None:
        raise ImportError("pdfplumber required: pip install pdfplumber")

    source_doc = Path(pdf_path).name
    department = _infer_department(source_doc)

    with pdfplumber.open(pdf_path) as pdf:
        full_text = "\n".join(page.extract_text() or "" for page in pdf.pages)

    # Normalize whitespace
    full_text = re.sub(r"\s+", " ", full_text).strip()
    words = full_text.split()

    chunks = []
    i = 0
    while i < len(words):
        chunk_words_slice = words[i: i + chunk_words]
        chunk_text = " ".join(chunk_words_slice)
        if len(chunk_words_slice) >= 20:  # skip tiny trailing fragments
            chunks.append({
                "text": chunk_text,
                "source_doc": source_doc,
                "chunk_index": len(chunks),
                "department": department,
                "chunk_id": f"{source_doc}:chunk:{len(chunks)}",
            })
        i += chunk_words - overlap_words

    return chunks
```

Create `knowledge/pipeline/embed.py`:
```python
from __future__ import annotations
import json
import time
import boto3
from typing import Any

def generate_embedding(bedrock_client: Any, text: str, region: str = "ap-south-1") -> list[float]:
    response = bedrock_client.invoke_model(
        modelId="amazon.titan-embed-text-v2:0",
        body=json.dumps({"inputText": text}),
    )
    body = json.loads(response["body"].read())
    return body["embedding"]  # 1536-dim float list

def embed_and_store_chunks(
    chunks: list[dict],
    bedrock_client: Any,
    pg_conn,           # psycopg2 / asyncpg connection
    dynamo_client: Any,
    dynamo_table: str = "voicebot-faq-metadata",
) -> int:
    """Embed chunks via Bedrock and write to Aurora + DynamoDB. Returns count stored."""
    stored = 0
    cursor = pg_conn.cursor()

    for chunk in chunks:
        embedding = generate_embedding(bedrock_client, chunk["text"])
        embedding_str = "[" + ",".join(str(v) for v in embedding) + "]"

        cursor.execute(
            """
            INSERT INTO faq_chunks (chunk_id, text, embedding, source_doc, department)
            VALUES (%s, %s, %s::vector, %s, %s)
            ON CONFLICT (chunk_id) DO UPDATE
              SET text=EXCLUDED.text, embedding=EXCLUDED.embedding
            """,
            (chunk["chunk_id"], chunk["text"], embedding_str, chunk["source_doc"], chunk["department"]),
        )

        dynamo_client.put_item(
            TableName=dynamo_table,
            Item={
                "department": {"S": chunk["department"]},
                "chunk_id":   {"S": chunk["chunk_id"]},
                "source_doc": {"S": chunk["source_doc"]},
                "text":       {"S": chunk["text"][:4096]},  # DynamoDB item size limit guard
            },
        )
        stored += 1

    pg_conn.commit()
    cursor.close()
    return stored
```

```python
#!/usr/bin/env python
"""
Monthly FAQ ingest runner.

Usage:
    python scripts/run_ingest.py --setup                     # Create Aurora table + DynamoDB table
    python scripts/run_ingest.py --pdf path/to/doc.pdf       # Ingest single PDF
    python scripts/run_ingest.py --pdf-dir knowledge/raw/    # Ingest all PDFs in directory
"""
import argparse, json, os, sys
sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))

import boto3
import psycopg2
from pathlib import Path
from knowledge.pipeline.ingest import extract_chunks_from_pdf
from knowledge.pipeline.embed import embed_and_store_chunks

def get_clients():
    region = os.getenv("AWS_REGION", "ap-south-1")
    bedrock = boto3.client("bedrock-runtime", region_name=region)
    dynamo  = boto3.client("dynamodb",        region_name=region)
    pg_conn = psycopg2.connect(os.environ["PG_CONN_STR"])
    return bedrock, dynamo, pg_conn

def setup(dynamo, pg_conn, table_name):
    schema = Path("knowledge/data/schemas/pg_schema.sql").read_text()
    cur = pg_conn.cursor()
    cur.execute(schema)
    pg_conn.commit()
    cur.close()
    print("Aurora schema applied.")

    dynamo_schema = json.loads(Path("knowledge/data/schemas/dynamo_metadata.json").read_text())
    dynamo_schema["TableName"] = table_name
    try:
        dynamo.create_table(**dynamo_schema)
        print(f"DynamoDB table {table_name} created.")
    except dynamo.exceptions.ResourceInUseException:
        print(f"DynamoDB table {table_name} already exists.")

def ingest_pdf(pdf_path, bedrock, dynamo, pg_conn, table_name):
    chunks = extract_chunks_from_pdf(pdf_path)
    count = embed_and_store_chunks(chunks, bedrock, pg_conn, dynamo, table_name)
    print(f"Stored {count} chunks from {pdf_path}")

if __name__ == "__main__":
    parser = argparse.ArgumentParser()
    parser.add_argument("--setup", action="store_true")
    parser.add_argument("--pdf", type=str)
    parser.add_argument("--pdf-dir", type=str)
    parser.add_argument("--table", default=os.getenv("DYNAMO_TABLE_NAME", "voicebot-faq-metadata"))
    args = parser.parse_args()

    bedrock, dynamo, pg_conn = get_clients()

    if args.setup:
        setup(dynamo, pg_conn, args.table)
    if args.pdf:
        ingest_pdf(args.pdf, bedrock, dynamo, pg_conn, args.table)
    if args.pdf_dir:
        for pdf in Path(args.pdf_dir).glob("*.pdf"):
            ingest_pdf(str(pdf), bedrock, dynamo, pg_conn, args.table)

    pg_conn.close()
```

```python
import asyncio
import json
import time
try:
    import psycopg2
    import psycopg2.extras
except ImportError:
    psycopg2 = None  # type: ignore

class AwsKnowledgeAdapter(KnowledgeAdapter):
    """Cloud production: Aurora PostgreSQL + pgvector hybrid search (cosine + BM25/ts_rank)."""

    # Hybrid SQL: combine vector cosine similarity and full-text ts_rank
    # Weights: 0.7 semantic + 0.3 keyword (voice queries are natural language — favor semantic)
    _HYBRID_SQL = """
        SELECT
            text,
            source_doc,
            (
                0.7 * (1 - (embedding <=> %(query_vec)s::vector))
                + 0.3 * ts_rank(to_tsvector('english', text), plainto_tsquery('english', %(query_text)s))
            ) AS score
        FROM faq_chunks
        WHERE
            embedding IS NOT NULL
            AND to_tsvector('english', text) @@ plainto_tsquery('english', %(query_text)s)
               OR (embedding <=> %(query_vec)s::vector) < 0.5
        ORDER BY score DESC
        LIMIT %(top_k)s
    """

    def __init__(self, pg_conn_str: str, bedrock_client: object) -> None:
        self._conn_str = pg_conn_str
        self._bedrock = bedrock_client

    def _get_connection(self):
        if psycopg2 is None:
            raise ImportError("psycopg2 required: pip install psycopg2-binary")
        return psycopg2.connect(self._conn_str)

    def _embed_query(self, query: str) -> list[float]:
        response = self._bedrock.invoke_model(
            modelId="amazon.titan-embed-text-v2:0",
            body=json.dumps({"inputText": query}),
        )
        body = json.loads(response["body"].read())
        return body["embedding"]

    def _run_hybrid_search(self, query_text: str, query_vec: list[float], top_k: int) -> list[dict]:
        conn = self._get_connection()
        try:
            cur = conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor)
            vec_str = "[" + ",".join(str(v) for v in query_vec) + "]"
            cur.execute(self._HYBRID_SQL, {
                "query_vec": vec_str,
                "query_text": query_text,
                "top_k": top_k,
            })
            rows = cur.fetchall()
            cur.close()
            return [dict(r) for r in rows]
        finally:
            conn.close()

    async def retrieve(self, query: str, top_k: int = 5) -> KnowledgeResult:
        t0 = time.monotonic()
        # Run blocking DB + embedding calls in thread pool to avoid blocking event loop
        query_vec, rows = await asyncio.to_thread(self._retrieve_sync, query, top_k)
        latency = (time.monotonic() - t0) * 1000
        return KnowledgeResult(
            chunks=[r["text"] for r in rows],
            sources=[r["source_doc"] for r in rows],
            search_latency_ms=latency,
        )

    def _retrieve_sync(self, query: str, top_k: int) -> tuple[list[float], list[dict]]:
        query_vec = self._embed_query(query)
        rows = self._run_hybrid_search(query, query_vec, top_k)
        return query_vec, rows
```
