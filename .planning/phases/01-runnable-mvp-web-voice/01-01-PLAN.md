    Write docker-compose.yml at C:/Coding/Enterprise-AI-Voice-Bot/docker-compose.yml.
    This is the local dev environment for two-tier deployment (Local + ECS only — no EC2 tier).

    Environment design:
    - orchestrator: builds from ./backend, port 8000, depends on Redis health
    - redis: image redis:7-alpine, port 6379, Redis sidecar pattern
    - AWS credential mount: ${HOME}/.aws:/root/.aws:ro
    - All config via env vars so same code runs locally and in ECS (config only changes between tiers)
    - USE_AWS_MOCKS=true by default for safe local dev (no accidental AWS calls)
    - AWS_REGION defaults to ap-south-1 (Mumbai — our ECS cluster region)

    ```yaml
    version: "3.9"

    services:
      orchestrator:
        build:
          context: ./backend
          dockerfile: Dockerfile
        ports:
          - "8000:8000"
        environment:
          USE_AWS_MOCKS: "${USE_AWS_MOCKS:-true}"
          AWS_REGION: "${AWS_REGION:-ap-south-1}"
          REDIS_URL: "redis://redis:6379"
          KNOWLEDGE_TABLE: "${KNOWLEDGE_TABLE:-voicebot-faq-knowledge}"
          CONVERSATIONS_TABLE: "${CONVERSATIONS_TABLE:-voicebot-conversations}"
          S3_FAQ_BUCKET: "${S3_FAQ_BUCKET:-voicebot-faq-documents}"
          BEDROCK_MODEL_ID: "${BEDROCK_MODEL_ID:-anthropic.claude-3-5-sonnet-20241022-v2:0}"
        volumes:
          - "${HOME}/.aws:/root/.aws:ro"
        depends_on:
          redis:
            condition: service_healthy
        healthcheck:
          test: ["CMD", "curl", "-f", "http://localhost:8000/health"]
          interval: 10s
          timeout: 5s
          retries: 3
          start_period: 30s

      redis:
        image: redis:7-alpine
        ports:
          - "6379:6379"
        healthcheck:
          test: ["CMD", "redis-cli", "ping"]
          interval: 5s
          timeout: 3s
          retries: 5
    ```
