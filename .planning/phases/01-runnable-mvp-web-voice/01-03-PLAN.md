```python
import os
import statistics
from collections import deque
from dataclasses import dataclass, field
from typing import Any, Optional

# Import PipelineResult lazily to avoid circular import
def _get_pipeline_result_class():
    from backend.app.orchestrator.pipeline import PipelineResult
    return PipelineResult


class LatencyBuffer:
    """Rolling in-process buffer for per-stage latency percentiles."""

    STAGES = ("asr", "rag", "llm", "tts")

    def __init__(self, maxlen: int = 1000) -> None:
        self._data: dict[str, deque] = {s: deque(maxlen=maxlen) for s in self.STAGES}

    def record(self, result: Any) -> None:
        """Accept a PipelineResult and append each stage's ms value."""
        total = 0.0
        for stage in self.STAGES:
            ms = getattr(result, f"{stage}_ms", 0.0)
            self._data[stage].append(ms)
            total += ms

    def percentiles(self, stage: str) -> dict[str, float]:
        """Return p50, p95, p99 for the given stage. Returns zeros if buffer empty."""
        if stage not in self.STAGES:
            raise ValueError(f"Unknown stage: {stage}. Valid: {self.STAGES}")
        data = sorted(self._data[stage])
        if not data:
            return {"p50": 0.0, "p95": 0.0, "p99": 0.0}
        n = len(data)
        return {
            "p50": data[int(n * 0.50)],
            "p95": data[min(int(n * 0.95), n - 1)],
            "p99": data[min(int(n * 0.99), n - 1)],
        }

    def all_percentiles(self) -> dict[str, dict[str, float]]:
        return {s: self.percentiles(s) for s in self.STAGES}


# Module-level singleton buffer — shared across requests within a process
_latency_buffer = LatencyBuffer()


def get_latency_buffer() -> LatencyBuffer:
    return _latency_buffer


def publish_stage_metrics(result: Any, trace_id: str, cw_client: Optional[Any] = None) -> None:
    """
    Record result in in-process buffer and publish to CloudWatch.
    Safe no-op when cw_client is None or USE_AWS_MOCKS=true.
    """
    _latency_buffer.record(result)

    use_mocks = os.getenv("USE_AWS_MOCKS", "true").lower() in {"1", "true", "yes"}
    if cw_client is None or use_mocks:
        return

    metric_data = [
        {"MetricName": "ASRLatency",  "Value": result.asr_ms, "Unit": "Milliseconds", "Dimensions": [{"Name": "TraceId", "Value": trace_id}]},
        {"MetricName": "RAGLatency",  "Value": result.rag_ms, "Unit": "Milliseconds", "Dimensions": [{"Name": "TraceId", "Value": trace_id}]},
        {"MetricName": "LLMLatency",  "Value": result.llm_ms, "Unit": "Milliseconds", "Dimensions": [{"Name": "TraceId", "Value": trace_id}]},
        {"MetricName": "TTSLatency",  "Value": result.tts_ms, "Unit": "Milliseconds", "Dimensions": [{"Name": "TraceId", "Value": trace_id}]},
    ]
    cw_client.put_metric_data(Namespace="voicebot/latency", MetricData=metric_data)
```

"""
Phase 1 E2E: verify RAG-augmented pipeline produces timing fields and /metrics works.
Run with: python -m pytest tests/e2e/test_phase1_roundtrip.py -v
"""
import asyncio
import pytest
from fastapi.testclient import TestClient
from backend.app.main import app
from backend.app.orchestrator.runtime import build_pipeline

def test_roundtrip_timing_fields_nonzero():
    """All four pipeline stage timings must be strictly positive after a voice turn.
    Using > 0.0 (not >= 0.0) to catch silent instrumentation failures where timers
    return zero due to short-circuit or exception swallowing.
    """
    pipeline = build_pipeline()
    result = asyncio.get_event_loop().run_until_complete(
        pipeline.run_roundtrip(b"fake-audio-bytes")
    )
    assert result.asr_ms > 0.0, f"asr_ms must be > 0, got {result.asr_ms}"
    assert result.rag_ms > 0.0, f"rag_ms must be > 0, got {result.rag_ms}"
    assert result.llm_ms > 0.0, f"llm_ms must be > 0, got {result.llm_ms}"
    assert result.tts_ms > 0.0, f"tts_ms must be > 0, got {result.tts_ms}"
    total = result.asr_ms + result.rag_ms + result.llm_ms + result.tts_ms
    assert total > 0.0, f"All stage timings zero — instrumentation broken. Total: {total}"

def test_metrics_endpoint_returns_valid_json():
    """GET /metrics returns 200 with expected stage keys."""
    client = TestClient(app)
    response = client.get("/metrics")
    assert response.status_code == 200
    data = response.json()
    for stage in ("asr", "rag", "llm", "tts"):
        assert stage in data, f"Missing stage '{stage}' in /metrics response"
        assert "p50" in data[stage]
        assert "p95" in data[stage]
        assert "p99" in data[stage]
